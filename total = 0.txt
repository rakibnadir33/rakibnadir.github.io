total = 0

print("Enter details for up to 10 items.")
print("Type 0 as price to stop early.\n")

for i in range(1, 11):
    price = float(input(f"Enter price of item {i}: "))
    
    # Allow early stop
    if price == 0:
        break

    quantity = int(input(f"Enter quantity of item {i}: "))
    total += price * quantity

# Apply rules
if total > 100:
    total *= 0.9  # 10% discount
elif total < 50:
    total += 5  # delivery fee

print(f"\nFinal total: ${total:.2f}")
